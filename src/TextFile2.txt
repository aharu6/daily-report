#Windows で WinError 5 を回避するビルド手順
#1) 実行中プロセスを止める（起動中だと dist\*.exe を削除できない）
Get-Process | Where-Object { $_.ProcessName -like '*業務見える化日誌*' } | Stop-Process -Force

#2) 既存EXEを削除（失敗しても継続）
Remove-Item .\dist\業務見える化日誌.exe -Force -ErrorAction SilentlyContinue

#3) クリーンビルド（uv 環境）venv環境は使用していないのでuv環境のみで書き出し
uv run flet pack main.py -i appicon\diary-left-svgrepo-com.ico --file-version 1.70.0.0 -n 業務見える化日誌
